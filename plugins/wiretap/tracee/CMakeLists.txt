include(WiresharkPlugin)

# Plugin name and version info (major minor micro extra)
set_module_info(tracee-wtap 0 0 1 0)

set(WIRETAP_SRC
	tracee-json.c
)

set(PLUGIN_FILES
	plugin.c
	${WIRETAP_SRC}
)

set_source_files_properties(
	${PLUGIN_FILES}
	PROPERTIES
	COMPILE_FLAGS "${WERROR_COMMON_FLAGS}"
)

register_plugin_files(plugin.c
	plugin_wtap
	"Add support for reading Tracee capture files"
	${WIRETAP_SRC}
)

add_wireshark_wiretap_plugin_library(tracee-wtap)

target_include_directories(tracee-wtap PRIVATE ${CMAKE_SOURCE_DIR}/wiretap)

target_link_libraries(tracee-wtap wiretap)

install_plugin(tracee-wtap wiretap)

file(GLOB WIRETAP_HEADERS RELATIVE "${CMAKE_CURRENT_SOURCE_DIR}" "*.h")
CHECKAPI(
	NAME
	  tracee-wtap
	SWITCHES
	SOURCES
	  ${WIRETAP_SRC}
	  ${WIRETAP_HEADERS}
)
